2. REPLACE with a real-data driven version:
   
   async function loadRealOutcomes(market, startDate, endDate) {
     const endpoint = {
       'btc': '/api/btc/candles?granularity=900',
       'spx': '/api/spx/candles?interval=1h',
       'mlb': '/api/mlb/events'
     }[market]
     
     const cacheKey = `marti_${market}_${startDate}_${endDate}`
     const cached = localStorage.getItem(cacheKey)
     if (cached) return JSON.parse(cached)
     
     const url = `${endpoint}&start=${startDate}&end=${endDate}`
     const res = await fetch(url)
     if (!res.ok) {
       throw new Error(`Failed to load ${market} data: ${res.status}`)
     }
     const outcomes = await res.json()
     localStorage.setItem(cacheKey, JSON.stringify(outcomes))
     return outcomes
   }

ASK MARTI: REAL ANTHROPIC API
============================================================

In AskMartyView, the current handleSend() function calls a demo stub. 
Replace with:

async function handleSend(message) {
  setMessages(prev => [...prev, {role: 'user', content: message}])
  setIsTyping(true)
  
  const res = await fetch('/api/ask-marti', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      messages: [...messages, {role: 'user', content: message}],
      context: buildAskMartyContext(currentState)  
      // include: mode, market, params, results, period, derived stats
    })
  })
  
  if (!res.ok) {
    setMessages(prev => [...prev, {
      role: 'assistant', 
      content: `Error: ${res.statusText}. Try again in a moment.`
    }])
    setIsTyping(false)
    return
  }
  
  // Handle streaming or just await full response
  const data = await res.json()
  setMessages(prev => [...prev, {role: 'assistant', content: data.content}])
  setIsTyping(false)
}
